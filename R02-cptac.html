Prepare visualisation

x <- assay(cptac_se) |> 
  as.data.frame() |> 
  rownames_to_column("Sequence") |> 
  full_join(as_tibble(rowData(cptac_se))) |> 
  filter(Proteins %in% prots) |> 
  pivot_longer(
    names_to = "rowname",
    values_to = "intensity", 
    starts_with("6")
  )
## Joining with `by = join_by(Sequence)`
cd <- colData(cptac_se) |> 
  as.data.frame() |> 
  rownames_to_column()
x <- full_join(x, 
          select(cd, -nNA)) 
## Joining with `by = join_by(rowname)`

Visualisation

ggplot(x, 
       aes(x = rowname,
           y = intensity,
           colour = group)) +
  geom_boxplot() +
  scale_y_log10() +
  facet_wrap(~ Proteins)
